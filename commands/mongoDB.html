# Queries

# Find

db.<collection>.find(<query>, <projection>)

coll.find( {}, { name: true });

# Sort

db.<collection>.find(<query>).sort(<sort order>)

coll = db.users;
coll.find().sort({ name: 1 });

# Limit & Skip
